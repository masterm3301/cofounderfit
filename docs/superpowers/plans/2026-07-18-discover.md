# Discover Profiles & Discover Projects Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add two paginated feed pages — `/discover/profiles` (signed-in, complete-profile only) and `/discover/projects` (public) — so users can browse people and projects instead of only reaching them by direct link.

**Architecture:** Two new data-access functions (`listProfiles`, `listProjects`) added to the existing `src/lib/profile.ts` / `src/lib/project.ts` modules, backed by a small shared pagination helper. Two new thin Server Component pages read `?page=N` from `searchParams`, call those functions, and render a grid of new card components. No new dependencies, no new routes outside `/discover/*`, no API routes — everything follows the existing pattern of Server Components calling `src/lib/` functions directly.

**Tech Stack:** Same as the foundation: Next.js 15 (App Router), TypeScript, Tailwind CSS, Prisma + Postgres (Neon), Vitest.

## Global Constraints

- Page size is a fixed constant, 20 items per page, for both feeds — no config, no user-facing page-size control.
- Pagination is offset-based via a `?page=N` query param (Prisma `skip`/`take`), not cursor-based.
- Both feeds order newest-first (`createdAt desc`).
- `/discover/profiles` requires a signed-in user with a complete profile (`requireCompleteProfile()`); `/discover/projects` is public, no guard.
- `listProfiles` excludes incomplete profiles (bio, skills, roleType, commitment not all set); `listProjects` has no completeness filter (every `Project` row is already complete by construction).
- No self-exclusion — a signed-in user's own profile/project can appear in the feed like anyone else's.
- Invalid or out-of-range `page` values (non-numeric, `<1`, or `>totalPages`) clamp to the nearest valid page — never a 404 or error page.
- An empty feed renders a plain "No profiles yet." / "No projects yet." message instead of an empty grid.

export const PAGE_SIZE = 20;

export function clampPage(page: number, totalPages: number): number {
  if (!Number.isFinite(page) || page < 1) return 1;
  return Math.min(Math.floor(page), totalPages);
}

import type { Profile, User } from "@prisma/client";
import { prisma } from "./db";
import { profileSchema, ProfileInput } from "./validation/profile";
import { PAGE_SIZE, clampPage } from "./pagination";

export async function getProfile(userId: string) {
  return prisma.profile.findUnique({ where: { userId } });
}

export function isProfileComplete(profile: Profile | null): boolean {
  if (!profile) return false;
  return Boolean(profile.bio) && profile.skills.length > 0 && Boolean(profile.roleType) && Boolean(profile.commitment);
}

export async function updateProfile(userId: string, input: ProfileInput) {
  const parsed = profileSchema.parse(input);
  return prisma.profile.update({ where: { userId }, data: parsed });
}

const COMPLETE_PROFILE_FILTER = {
  bio: { not: null },
  roleType: { not: null },
  commitment: { not: null },
  skills: { isEmpty: false },
};

export async function listProfiles(
  page: number
): Promise<{ profiles: (Profile & { user: User })[]; totalPages: number }> {
  const total = await prisma.profile.count({ where: COMPLETE_PROFILE_FILTER });
  const totalPages = Math.max(1, Math.ceil(total / PAGE_SIZE));
  const currentPage = clampPage(page, totalPages);

  const profiles = await prisma.profile.findMany({
    where: COMPLETE_PROFILE_FILTER,
    include: { user: true },
    orderBy: { user: { createdAt: "desc" } },
    skip: (currentPage - 1) * PAGE_SIZE,
    take: PAGE_SIZE,
  });

  return { profiles, totalPages };
}
```

import type { Project, User } from "@prisma/client";
import { prisma } from "./db";
import { projectSchema, ProjectInput } from "./validation/project";
import { PAGE_SIZE, clampPage } from "./pagination";

export async function getProject(userId: string) {
  return prisma.project.findUnique({ where: { ownerId: userId } });
}

export async function getProjectById(projectId: string) {
  return prisma.project.findUnique({ where: { id: projectId }, include: { owner: true } });
}

export async function createProject(userId: string, input: ProjectInput) {
  const existing = await getProject(userId);
  if (existing) {
    throw new Error("You already have an active project. Edit it instead of creating a new one.");
  }
  const parsed = projectSchema.parse(input);
  return prisma.project.create({ data: { ...parsed, ownerId: userId } });
}

export async function updateProject(userId: string, input: ProjectInput) {
  const parsed = projectSchema.parse(input);
  return prisma.project.update({ where: { ownerId: userId }, data: parsed });
}

export async function deleteProject(userId: string) {
  await prisma.project.delete({ where: { ownerId: userId } });
}

export async function listProjects(
  page: number
): Promise<{ projects: (Project & { owner: User })[]; totalPages: number }> {
  const total = await prisma.project.count();
  const totalPages = Math.max(1, Math.ceil(total / PAGE_SIZE));
  const currentPage = clampPage(page, totalPages);

  const projects = await prisma.project.findMany({
    include: { owner: true },
    orderBy: { createdAt: "desc" },
    skip: (currentPage - 1) * PAGE_SIZE,
    take: PAGE_SIZE,
  });

  return { projects, totalPages };
}
```

import { listProfiles } from "@/lib/profile";
import { clampPage } from "@/lib/pagination";
import { ProfileCard } from "@/components/ProfileCard";

export default async function DiscoverProfilesPage({
  searchParams,
}: {
  searchParams: Promise<{ page?: string }>;
}) {
  await requireCompleteProfile();

  const { page } = await searchParams;
  const { profiles, totalPages } = await listProfiles(Number(page));

            <ProfileCard key={profile.userId} profile={profile} />

        </div>

        {currentPage > 1 && (

          </a>
        )}
        <span>
          Page {currentPage} of {totalPages}
        </span>
        {currentPage < totalPages && (
          <a href={`/discover/profiles?page=${currentPage + 1}`} className="underline">
            Next
          </a>
        )}
      </div>
    </main>
  );
}
